> ```go
> package main
> 
> import "fmt"
> 
> 
> func set(arr []int) []int {
> 	removeRepeat := make(map[int]int)
> 
> 	for i := 0; i < len(arr); i++ {
> 		removeRepeat[arr[i]] = 0
> 	}
> 	slice := make([]int, 0, len(arr))
> 
> 	for k, _ := range removeRepeat {
> 		slice = append(slice, k)
> 	}
> 
> 	return slice
> }
> 
> func main() {
> 
> 	slice := make([]int, 0, 10)
> 	slice = append(slice, 1, 2, 3, 4, 4, 5, 6, 6, 7)
> 	//	打印slice
> 	slice = set(slice)
> 	for i := 0; i < len(slice); i++ {
> 		fmt.Println(slice[i])
> 	}
> 
> }
> 
> ```
